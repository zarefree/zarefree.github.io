安装 Golang ​

  • 解压
sh
tar -xzvf <package-name>
  • 剪切
sh
sudo mv go/ /usr/local/
  • 配置环境变量
sh
vim ~/.bashrc

export GOROOT=/usr/local/go
export GOPATH=$HOME/go
export PATH=$PATH:$GOROOT/bin:$GOPATH/bin

source ~/.bashrc
