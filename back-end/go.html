密钥生成 ​

go
package main

import (
	"crypto/rand"
	"crypto/rsa"
	"crypto/x509"
	"encoding/pem"
	"os"
)

func RsaGenKey(bits int) error {
	privateKey, err := rsa.GenerateKey(rand.Reader, bits)
	if err != nil {
		return err
	}

	derStream := x509.MarshalPKCS1PrivateKey(privateKey)
	block := pem.Block{
		Type:  "RSA PRIVATE KEY", //"RSA PRIVATE KEY",
		Bytes: derStream,
	}

	privFile, err := os.Create("private.pem")
	if err != nil {
		return err
	}

	err = pem.Encode(privFile, &block)
	if err != nil {
		return err
	}

	defer privFile.Close()

	publicKey := privateKey.PublicKey
	derPkix, err := x509.MarshalPKIXPublicKey(&publicKey)
	if err != nil {
		return err
	}

	block = pem.Block{
		Type:  "RSA PUBLIC KEY", //"PUBLIC KEY",
		Bytes: derPkix,
	}

	pubFile, err := os.Create("public.pem")
	if err != nil {
		return err
	}

	err = pem.Encode(pubFile, &block)
	if err != nil {
		panic(err)
	}

	defer pubFile.Close()

	return nil
}

func main() {
	RsaGenKey(2048) // 1024 2048 3072 4096
}

var ErrDecryption = rsa.ErrDecryption

func Encrypt(plaintText string) string {
	block, _ := pem.Decode([]byte(publicKey))
	pkixPublicKey, _ := x509.ParsePKIXPublicKey(block.Bytes)
	cipherText, _ := rsa.EncryptPKCS1v15(rand.Reader, pkixPublicKey.(*rsa.PublicKey), []byte(plaintText))
	return base64.StdEncoding.EncodeToString(cipherText)
}

func Decrypt(cipherText string) (string, error) {
	block, _ := pem.Decode([]byte(privateKey))
	pkixPrivateKey, err := x509.ParsePKCS1PrivateKey(block.Bytes)
	if err != nil {
		return "", err
	}

	decodedBytes, err := base64.StdEncoding.DecodeString(cipherText)
	if err != nil {
		return "", err
	}

	plaintText, err := rsa.DecryptPKCS1v15(rand.Reader, pkixPrivateKey, decodedBytes)
	if err != nil {
		return "", err
	}

	return string(plaintText), nil
}

func main() {
	encryptPwd := Encrypt("123456")
	decryptPwd, _ := Decrypt(encryptPwd)
	fmt.Println(encryptPwd)
	fmt.Println(decryptPwd)
}

Pbkdf2 密码哈希算法 ​

go
package main

import (
	"crypto/sha512"
	"encoding/hex"
	"fmt"

	"golang.org/x/crypto/pbkdf2"
)

func pbkdf2Key(rawPwd string) []byte {
	// 128 位
	salt := "ZKRq5Nz1z5ie5g5ctUvCESix4H7UxakkiWqnlG106waBPC4JMDT3cmv4r1h4wNJVS5RpWAQ94PWx5s8b81l5alLE6yVhRphQXn2ijunZ4ym4AGQuRTG3jzfyfimpL4TB"
	// 迭代次数
	iter := 1000000
	// keyLen
	keyLen := 64
	// HashFunction
	hashFunc := sha512.New

	return pbkdf2.Key([]byte(rawPwd), []byte(salt), iter, keyLen, hashFunc)
}

func Encode(rawPwd string) string {
	return hex.EncodeToString(pbkdf2Key(rawPwd))
}

func Verify(rawPwd string, encodedPwd string) bool {
	return encodedPwd == hex.EncodeToString(pbkdf2Key(rawPwd))
}

func main() {
	encodedPwd := Encode("123456")
	check := Verify("123456", encodedPwd)
	fmt.Println(encodedPwd)
	fmt.Println(check) // true
}

Gorm many2many ​

go
package main

import (
	"fmt"
	"time"

	"gorm.io/driver/mysql"
	"gorm.io/gorm"
)

var DB *gorm.DB

func InitMysql() {
	dsn := "root:qh_0423@tcp(127.0.0.1:3307)/many2many"
	config := mysql.Config{
		DSN:                       dsn,
		SkipInitializeWithVersion: false,
	}

	db, err := gorm.Open(mysql.New(config), &gorm.Config{})
	if err != nil {
		panic(err)
	}
	sd, err := db.DB()
	if err != nil {
		return
	}

	sd.SetMaxIdleConns(10)
	sd.SetMaxOpenConns(50)
	sd.SetConnMaxLifetime(time.Hour)

	DB = db
}

type Student struct {
	Id      uint     `json:"id" gorm:"primaryKey;autoIncrement"`
	Name    string   `json:"name" gorm:"type:varchar(255);not null"`
	Courses []Course `json:"courses" gorm:"many2many:student_course"`
}

func (Student) TableName() string {
	return "student"
}

type Course struct {
	Id   uint   `json:"id" gorm:"primaryKey;autoIncrement"`
	Name string `json:"name" gorm:"type:varchar(255);not null"`
}

func (Course) TableName() string {
	return "course"
}

type StudentCourse struct {
	Id        uint `json:"id" gorm:"primaryKey;autoIncrement"`
	StudentId uint `json:"studentId" gorm:"type:bigint;index:idx_student_id;not null"`
	CourseId  uint `json:"courseId" gorm:"type:bigint;index:idx_course_id;not null"`
}

func (StudentCourse) TableName() string {
	return "student_course"
}

func main() {
	InitMysql()
	DB.AutoMigrate(&Student{}, &Course{}, &StudentCourse{})
	var student = []Student{
		{Id: 1, Name: "张三"},
		{Id: 2, Name: "李四"},
	}
	DB.Create(&student)
	var course = []Course{
		{Id: 1, Name: "语文"},
		{Id: 2, Name: "数学"},
		{Id: 3, Name: "英语"},
		{Id: 4, Name: "物理"},
		{Id: 5, Name: "地理"},
		{Id: 6, Name: "生物"},
		{Id: 7, Name: "化学"},
	}
	DB.Create(&course)
	var studentCourse = []StudentCourse{
		{StudentId: 1, CourseId: 1},
		{StudentId: 1, CourseId: 2},
		{StudentId: 1, CourseId: 3},
		{StudentId: 2, CourseId: 5},
		{StudentId: 2, CourseId: 6},
		{StudentId: 2, CourseId: 7},
	}
	DB.Create(&studentCourse)

	var s []Student
	err := DB.Model(&Student{}).Preload("Courses").Find(&s).Error
	if err != nil {
		fmt.Println(err)
	}

	fmt.Println(s)
}
