package main

import (
	"fmt"
	"time"

	"gorm.io/driver/mysql"
	"gorm.io/gorm"
)

var DB *gorm.DB

func InitMysql() {
	dsn := "root:qh_0423@tcp(127.0.0.1:3307)/many2many"
	config := mysql.Config{
		DSN:                       dsn,
		SkipInitializeWithVersion: false,
	}

	db, err := gorm.Open(mysql.New(config), &gorm.Config{})
	if err != nil {
		panic(err)
	}
	sd, err := db.DB()
	if err != nil {
		return
	}

	sd.SetMaxIdleConns(10)
	sd.SetMaxOpenConns(50)
	sd.SetConnMaxLifetime(time.Hour)

	DB = db
}

type Student struct {
	Id      uint     `json:"id" gorm:"primaryKey;autoIncrement"`
	Name    string   `json:"name" gorm:"type:varchar(255);not null"`
	Courses []Course `json:"courses" gorm:"many2many:student_course"`
}

func (Student) TableName() string {
	return "student"
}

type Course struct {
	Id   uint   `json:"id" gorm:"primaryKey;autoIncrement"`
	Name string `json:"name" gorm:"type:varchar(255);not null"`
}

func (Course) TableName() string {
	return "course"
}

type StudentCourse struct {
	Id        uint `json:"id" gorm:"primaryKey;autoIncrement"`
	StudentId uint `json:"studentId" gorm:"type:bigint;index:idx_student_id;not null"`
	CourseId  uint `json:"courseId" gorm:"type:bigint;index:idx_course_id;not null"`
}

func (StudentCourse) TableName() string {
	return "student_course"
}
